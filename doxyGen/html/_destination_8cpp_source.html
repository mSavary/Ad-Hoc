47 Destination::Destination(IPv6 *ip, int metric) {
48  mIP = ip;
49  mMetric = metric;
50  mState = NONE;
51  mDel = true;
52  mIo = new boost::asio::io_service();
53  if (mMetric == 1) {
54  mTimer = new boost::asio::deadline_timer(*mIo,
55  boost::posix_time::seconds(NEIGHB_HOLD_TIME));

57  mTimer = new boost::asio::deadline_timer(*mIo,
58  boost::posix_time::seconds(TOP_HOLD_TIME));

62 }
